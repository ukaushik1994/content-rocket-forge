**Current flow:**
```

```

**New flow:**
```

```

**Change the existing cannibalization block (~line 504-522) to:**

```ts
// SB-20: Cannibalization prevention — BLOCKING check
if (existingTargeting && existingTargeting.length >= 2) {
```

## PHASE B: The Content Improvement Tool (M1-1)

This is the single highest-impact remaining feature. Every user who creates content will eventually want to improve it. Right now they can only create new — there's no "make my existing article better."

### Backend: `supabase/functions/enhanced-ai-chat/content-action-tools.ts`

**Add new tool definition to TOOL_DEFINITIONS array:**

```ts
{
  name: 'improve_content',
  description: 'Improve an existing content item using AI — rewrites, expands, updates, or optimizes it in-place while preserving the original as a version',
  parameters: {
    type: 'object',
    properties: {
      content_id: { type: 'string', description: 'ID of the content item to improve' },
      instruction: { type: 'string', description: 'What to improve: "rewrite intro", "add more examples", "improve SEO", "make more conversational", "update statistics", "shorten by 30%"' },
      scope: { type: 'string', enum: ['full', 'intro', 'conclusion', 'specific_section'], description: 'Which part to improve. Default: full' }
    },
    required: ['content_id', 'instruction']
  }
}
```

**Add handler function:**

```ts
async function handleImproveContent(toolArgs: any, userId: string, supabase: any, aiProxy: Function) {
  // 1. Fetch existing content
  const { data: content } = await supabase.from('content_items')
    .select('*')
    .eq('id', toolArgs.content_id)
    .eq('user_id', userId)
    .single();

  if (!content) return JSON.stringify({ success: false, message: 'Content not found' });

  // 2. Snapshot current version BEFORE changing anything
  await supabase.from('content_versions').insert({
    content_id: content.id,
    user_id: userId,
    title: content.title,
    content: content.content,
    meta_title: content.meta_title,
    meta_description: content.meta_description,
    seo_score: content.seo_score,
    change_source: 'ai_improvement',
    change_description: `Before improvement: ${toolArgs.instruction}`
  });

  // 3. Build improvement prompt
  const improvementPrompt = `You are improving an existing article. Here is the current content:

TITLE: ${content.title}
KEYWORD: ${content.main_keyword || 'not set'}
CURRENT SEO SCORE: ${content.seo_score || 'not scored'}

CONTENT:
${content.content}

INSTRUCTION: ${toolArgs.instruction}
SCOPE: ${toolArgs.scope || 'full'}

Rules:
- Maintain the same voice and style
- Keep all existing internal links
- If improving SEO: add the keyword naturally, improve headings, add FAQ section if missing
- If shortening: preserve key points and structure, remove filler
- If expanding: add real value (examples, data, actionable tips), not fluff
- Return ONLY the improved HTML content, no commentary`;

  // 4. Call AI for improvement
  const improved = await callAiProxyWithRetry(supabase, userId, {
    messages: [{ role: 'user', content: improvementPrompt }],
    max_tokens: 4000
  });

  const improvedContent = improved.choices?.[0]?.message?.content;
  if (!improvedContent) return JSON.stringify({ success: false, message: 'AI did not return improved content' });

  // 5. Rescore and save
  const newScore = calculateBasicSeoScore(improvedContent, content.main_keyword, content.meta_title, content.meta_description);

  await supabase.from('content_items').update({
    content: improvedContent,
    seo_score: newScore,
    updated_at: new Date().toISOString()
  }).eq('id', content.id).eq('user_id', userId);

  const scoreDelta = newScore - (content.seo_score || 0);

  return JSON.stringify({
    success: true,
    message: `✅ Improved "${content.title}". SEO score: ${content.seo_score || 0} → ${newScore} (${scoreDelta >= 0 ? '+' : ''}${scoreDelta}). Previous version saved — you can restore it from content versions.`,
    content_id: content.id,
    old_score: content.seo_score,
    new_score: newScore
  });
}
```

**Add to tool registry in `tools.ts`:** Register `improve_content` in the CONTENT category so it's available when content intent is detected.

**Frontend:** No new components needed — the tool works through chat. User says "improve my article about X" or "make my SEO article shorter" and the AI calls the tool.

**Why this matters:** This is the #1 thing users will try after creating their first few articles. "Can you make this better?" is a natural request. Without it, the AI chat can only create, never refine — which makes it feel disposable.

### M1-19: AI-Driven Meta Descriptions (Not Just Truncation)

#### Backend: `supabase/functions/enhanced-ai-chat/content-action-tools.ts`

The current meta description generation (~line 791) is:
```ts
const autoMetaDesc = plainText.length > 155 ? plainText.substring(0, 152) + '...' : plainText;
```

This is terrible — it just chops the first 155 characters of article text. Replace with AI-generated meta:

```ts
// Generate meta title and description via AI (not truncation)
let autoMetaTitle = autoTitle;
let autoMetaDesc = '';

try {
  const metaResponse = await callAiProxyWithRetry(supabase, userId, {
    messages: [{
      role: 'user',
      content: `Generate a meta title and meta description for this article.

TITLE: ${autoTitle}
KEYWORD: ${toolArgs.keyword}
FIRST 500 CHARS: ${plainText.substring(0, 500)}

Rules for meta TITLE (max 60 chars):
- Include the primary keyword near the start
- Use a power word (Ultimate, Complete, Essential, Proven)
- Make it specific — include a number or year if relevant

Rules for meta DESCRIPTION (max 155 chars):
- Write a HOOK, not a summary — create curiosity
- Start with an action verb or bold claim
- Include the keyword naturally
- End with implied value ("...here's how" / "...that actually work")

Return ONLY valid JSON: {"meta_title": "...", "meta_description": "..."}`
    }],
    max_tokens: 200
  });

  const metaJson = JSON.parse(metaResponse.choices?.[0]?.message?.content || '{}');
  if (metaJson.meta_title) autoMetaTitle = metaJson.meta_title.substring(0, 60);
  if (metaJson.meta_description) autoMetaDesc = metaJson.meta_description.substring(0, 155);
} catch {
  // Fallback to truncation if AI fails
  autoMetaDesc = plainText.length > 155 ? plainText.substring(0, 152) + '...' : plainText;
}

if (!autoMetaDesc) {
  autoMetaDesc = plainText.length > 155 ? plainText.substring(0, 152) + '...' : plainText;
}
```

**Frontend:** No changes — meta fields are already displayed in wizard and content detail.

**Why this matters:** Meta descriptions are what users see in Google results. Truncated text = 0% click-through improvement. An AI hook = measurable traffic difference. This is one of those "small change, big impact" fixes.

```ts
async function handleMonthlySummary(toolArgs: any, userId: string, supabase: any) {
  const targetMonth = toolArgs.month || new Date().toISOString().slice(0, 7); // YYYY-MM
  const startDate = `${targetMonth}-01`;
  const endDate = new Date(new Date(startDate).setMonth(new Date(startDate).getMonth() + 1)).toISOString().slice(0, 10);

  // Parallel queries
  const [contentCreated, contentPublished, keywordsAdded, proposalsAccepted, calendarCompleted, emailsSent] = await Promise.all([
    supabase.from('content_items').select('id, title, seo_score, status', { count: 'exact' })
      .eq('user_id', userId).gte('created_at', startDate).lt('created_at', endDate),
    supabase.from('content_items').select('id, title, seo_score')
      .eq('user_id', userId).eq('status', 'published').gte('updated_at', startDate).lt('updated_at', endDate),
    supabase.from('keywords').select('id, keyword', { count: 'exact' })
      .eq('user_id', userId).gte('created_at', startDate).lt('created_at', endDate),
    supabase.from('ai_strategy_proposals').select('id', { count: 'exact' })
      .eq('user_id', userId).eq('status', 'scheduled').gte('updated_at', startDate).lt('updated_at', endDate),
    supabase.from('content_calendar').select('id', { count: 'exact' })
      .eq('user_id', userId).eq('status', 'completed').gte('scheduled_date', startDate).lt('scheduled_date', endDate),
    supabase.from('email_campaigns').select('id', { count: 'exact' })
      .eq('user_id', userId).eq('status', 'sent').gte('updated_at', startDate).lt('updated_at', endDate)
  ]);

  const avgSeo = contentCreated.data?.length
    ? Math.round(contentCreated.data.reduce((sum: number, c: any) => sum + (c.seo_score || 0), 0) / contentCreated.data.length)
    : 0;

  const topContent = [...(contentCreated.data || [])].sort((a: any, b: any) => (b.seo_score || 0) - (a.seo_score || 0)).slice(0, 3);

  return JSON.stringify({
    success: true,
    month: targetMonth,
    summary: {
      content_created: contentCreated.count || 0,
      content_published: contentPublished.data?.length || 0,
      avg_seo_score: avgSeo,
      keywords_added: keywordsAdded.count || 0,
      proposals_accepted: proposalsAccepted.count || 0,
      calendar_items_completed: calendarCompleted.count || 0,
      emails_sent: emailsSent.count || 0,
      top_content: topContent.map((c: any) => ({ title: c.title, seo_score: c.seo_score }))
    },
    message: `📊 **${targetMonth} Summary**: ${contentCreated.count || 0} articles created, ${contentPublished.data?.length || 0} published, avg SEO ${avgSeo}/100, ${keywordsAdded.count || 0} keywords added, ${proposalsAccepted.count || 0} proposals acted on.`
  });
}
```
