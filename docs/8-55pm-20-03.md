```ts
async function handleImproveContent(
  toolArgs: any,
  userId: string,
  supabase: any,
  callAiProxy: Function
): Promise<string> {
  // 1. Fetch the content item
  const { data: content, error: fetchErr } = await supabase
    .from('content_items')
    .select('*')
    .eq('id', toolArgs.content_id)
    .eq('user_id', userId)
    .single();

  if (fetchErr || !content) {
    return JSON.stringify({
      success: false,
      message: 'Content not found or you don\'t have access to it.'
    });
  }

  // 2. Snapshot the current version BEFORE changing anything
  try {
    await supabase.from('content_versions').insert({
      content_id: content.id,
      user_id: userId,
      title: content.title,
      content: content.content,
      meta_title: content.meta_title,
      meta_description: content.meta_description,
      seo_score: content.seo_score,
      change_source: 'ai_improvement',
      change_description: `Before improvement: ${toolArgs.instruction}`
    });
  } catch (versionErr) {
    console.warn('Version snapshot failed (non-blocking):', versionErr);
  }

  // 3. Build the improvement prompt
  const scope = toolArgs.scope || 'full';
  const prompt = `You are improving an existing article. Here is the current content:

TITLE: ${content.title}
KEYWORD: ${content.main_keyword || 'not specified'}
CURRENT SEO SCORE: ${content.seo_score || 'not scored'}

CONTENT:
${content.content}

---

IMPROVEMENT INSTRUCTION: ${toolArgs.instruction}
SCOPE: ${scope}

Rules:
- Maintain the same voice, style, and formatting approach unless the instruction says otherwise
- Keep all existing internal links (<a> tags) intact — do not remove them
- If improving SEO: ensure the keyword appears in the first 100 words, improve heading structure, add FAQ section if missing
- If shortening: preserve all key points, remove filler and redundant sentences
- If expanding: add real value (concrete examples, data points, actionable tips) — never add fluff or padding
- If the scope is "intro" or "conclusion", only rewrite that section — return the FULL article with only that section changed
- Return ONLY the improved HTML content. No commentary, no explanations, no markdown — just the HTML that replaces the current content.`;

  // 4. Call AI for improvement
  let improvedContent: string;
  try {
    const aiResponse = await callAiProxy(supabase, userId, {
      messages: [{ role: 'user', content: prompt }],
      max_tokens: 4000
    });
    improvedContent = aiResponse.choices?.[0]?.message?.content || '';
  } catch (aiErr: any) {
    return JSON.stringify({
      success: false,
      message: `AI improvement failed: ${aiErr.message || 'Unknown error'}. Your original content is unchanged.`
    });
  }

  if (!improvedContent || improvedContent.trim().length < 50) {
    return JSON.stringify({
      success: false,
      message: 'AI returned empty or too-short content. Your original content is unchanged. Try a more specific instruction.'
    });
  }

  // 5. Calculate new SEO score
  const newScore = calculateBasicSeoScore(
    improvedContent,
    content.main_keyword,
    content.meta_title,
    content.meta_description
  );

  // 6. Save the improved content
  const { error: updateErr } = await supabase
    .from('content_items')
    .update({
      content: improvedContent,
      seo_score: newScore,
      updated_at: new Date().toISOString()
    })
    .eq('id', content.id)
    .eq('user_id', userId);

  if (updateErr) {
    return JSON.stringify({
      success: false,
      message: `Failed to save improved content: ${updateErr.message}. Your previous version was saved — you can restore it.`
    });
  }

  const scoreDelta = newScore - (content.seo_score || 0);
  const oldWords = (content.content || '').replace(/<[^>]+>/g, ' ').split(/\s+/).filter(Boolean).length;
  const newWords = improvedContent.replace(/<[^>]+>/g, ' ').split(/\s+/).filter(Boolean).length;
  const wordDelta = newWords - oldWords;

  return JSON.stringify({
    success: true,
    message: `Improved "${content.title}".\n\n` +
      `**SEO:** ${content.seo_score || 0} → ${newScore} (${scoreDelta >= 0 ? '+' : ''}${scoreDelta})\n` +
      `**Words:** ${oldWords.toLocaleString()} → ${newWords.toLocaleString()} (${wordDelta >= 0 ? '+' : ''}${wordDelta})\n\n` +
      `Previous version saved automatically — use "restore content version" to undo.`,
    content_id: content.id,
    old_score: content.seo_score,
    new_score: newScore
  });
}
```

```ts
async function handleReformatContent(
  toolArgs: any,
  userId: string,
  supabase: any,
  callAiProxy: Function
): Promise<string> {
  // Fetch content (same as improve_content)
  const { data: content, error: fetchErr } = await supabase
    .from('content_items')
    .select('*')
    .eq('id', toolArgs.content_id)
    .eq('user_id', userId)
    .single();

  if (fetchErr || !content) {
    return JSON.stringify({ success: false, message: 'Content not found or access denied.' });
  }

  // Snapshot version (same as improve_content)
  try {
    await supabase.from('content_versions').insert({
      content_id: content.id,
      user_id: userId,
      title: content.title,
      content: content.content,
      meta_title: content.meta_title,
      meta_description: content.meta_description,
      seo_score: content.seo_score,
      change_source: 'reformat',
      change_description: `Before reformat: ${toolArgs.target}`
    });
  } catch { /* non-blocking */ }

  // Target-specific instructions
  const targetInstructions: Record<string, string> = {
    shorter: 'Reduce the content by 30-40% while keeping ALL key points, conclusions, and data. Remove filler sentences, redundant explanations, and wordy phrases. Tighten every paragraph.',
    longer: 'Expand the content by 30-50% with REAL value: concrete examples, relevant data points, actionable tips, brief case references, or additional context. Do NOT add fluff, filler, or restate existing points in different words.',
    more_casual: 'Rewrite in a warm, conversational tone. Use contractions (you\'re, it\'s, don\'t), shorter sentences, direct address ("you"), occasional rhetorical questions, and a friendly approachable voice. Keep all information intact.',
    more_formal: 'Rewrite in a professional, authoritative tone. Remove all contractions, slang, and casual language. Use precise terminology, complete sentences, and a structured logical flow. Keep all information intact.',
    add_bullets: 'Convert long paragraphs into bulleted or numbered lists where the content is a series of points, steps, or items. Keep narrative/explanatory sections as paragraphs. Use bullets for scannability.',
    simplify: 'Rewrite at a 6th-grade reading level. Use short sentences (max 15 words). Use common everyday words instead of jargon. Break complex ideas into simple steps. Preserve ALL information — just make it easier to understand.'
  };

  let instruction = targetInstructions[toolArgs.target] || `Reformat the content: ${toolArgs.target}`;
  if (toolArgs.target_word_count) {
    instruction += ` Target approximately ${toolArgs.target_word_count} words.`;
  }

  // Build prompt
  const prompt = `Reformat this article according to the instruction below.

TITLE: ${content.title}
KEYWORD: ${content.main_keyword || 'not specified'}

CONTENT:
${content.content}

---

REFORMAT INSTRUCTION: ${instruction}

Rules:
- Keep all internal links (<a> tags) intact
- Keep the same heading structure unless the instruction requires changing it
- Return ONLY the reformatted HTML content. No commentary.`;

  // Call AI
  let reformatted: string;
  try {
    const aiResponse = await callAiProxy(supabase, userId, {
      messages: [{ role: 'user', content: prompt }],
      max_tokens: 4000
    });
    reformatted = aiResponse.choices?.[0]?.message?.content || '';
  } catch (aiErr: any) {
    return JSON.stringify({ success: false, message: `Reformat failed: ${aiErr.message}. Original content unchanged.` });
  }

  if (!reformatted || reformatted.trim().length < 50) {
    return JSON.stringify({ success: false, message: 'AI returned empty content. Original unchanged. Try again.' });
  }

  // Rescore and save
  const newScore = calculateBasicSeoScore(reformatted, content.main_keyword, content.meta_title, content.meta_description);

  await supabase.from('content_items').update({
    content: reformatted,
    seo_score: newScore,
    updated_at: new Date().toISOString()
  }).eq('id', content.id).eq('user_id', userId);

  const oldWords = (content.content || '').replace(/<[^>]+>/g, ' ').split(/\s+/).filter(Boolean).length;
  const newWords = reformatted.replace(/<[^>]+>/g, ' ').split(/\s+/).filter(Boolean).length;

  return JSON.stringify({
    success: true,
    message: `Reformatted "${content.title}" → ${toolArgs.target}.\n\n` +
      `**Words:** ${oldWords.toLocaleString()} → ${newWords.toLocaleString()}\n` +
      `**SEO:** ${content.seo_score || 0} → ${newScore}\n\n` +
      `Previous version saved — use "restore content version" to undo.`,
    content_id: content.id
  });
}
```

**Backend:** `supabase/functions/enhanced-ai-chat/content-action-tools.ts`

```ts
// AI-generated meta title and description (replaces truncation)
let autoMetaTitle = autoTitle;
let autoMetaDesc = '';

try {
  const metaResponse = await callAiProxyWithRetry(supabase, userId, {
    messages: [{
      role: 'user',
      content: `Generate a meta title and meta description for this article.

TITLE: ${autoTitle}
KEYWORD: ${toolArgs.keyword || ''}
ARTICLE OPENING: ${plainText.substring(0, 500)}

Rules for meta TITLE (maximum 60 characters):
- Include the primary keyword near the start
- Use a power word (Ultimate, Complete, Essential, Proven, Simple)
- Be specific — include a number or year if it fits naturally
- Must be under 60 characters total

Rules for meta DESCRIPTION (maximum 155 characters):
- Write a HOOK that creates curiosity — NOT a summary of the article
- Start with an action verb or a bold claim
- Include the keyword naturally, not forced
- End with implied value: "here's how", "that actually work", "you need to know"
- Must be under 155 characters total

Return ONLY valid JSON with no other text:
{"meta_title": "your title here", "meta_description": "your description here"}`
    }],
    max_tokens: 200
  });

  const raw = metaResponse.choices?.[0]?.message?.content || '';
  // Extract JSON — handle cases where AI wraps in markdown code blocks
  const jsonMatch = raw.match(/\{[\s\S]*\}/);
  if (jsonMatch) {
    const parsed = JSON.parse(jsonMatch[0]);
    if (parsed.meta_title && parsed.meta_title.length <= 60) {
      autoMetaTitle = parsed.meta_title;
    }
    if (parsed.meta_description && parsed.meta_description.length <= 155) {
      autoMetaDesc = parsed.meta_description;
    }
  }
} catch (metaErr) {
  console.warn('AI meta generation failed, falling back to truncation:', metaErr);
}

// Fallback to truncation if AI failed or returned nothing
if (!autoMetaDesc) {
  autoMetaDesc = plainText.length > 155 ? plainText.substring(0, 152) + '...' : plainText;
}
if (!autoMetaTitle || autoMetaTitle.length > 60) {
  autoMetaTitle = autoTitle.length > 60 ? autoTitle.substring(0, 57) + '...' : autoTitle;
}
```

```ts
async function handleMonthlySummary(
  toolArgs: any,
  userId: string,
  supabase: any
): Promise<string> {
  const targetMonth = toolArgs.month || new Date().toISOString().slice(0, 7); // YYYY-MM
  const startDate = `${targetMonth}-01`;
  const nextMonth = new Date(startDate);
  nextMonth.setMonth(nextMonth.getMonth() + 1);
  const endDate = nextMonth.toISOString().slice(0, 10);

  // Run all queries in parallel for speed
  const [created, published, keywords, proposals, calendar] = await Promise.all([
    supabase.from('content_items')
      .select('id, title, seo_score, status', { count: 'exact' })
      .eq('user_id', userId)
      .gte('created_at', startDate)
      .lt('created_at', endDate),

    supabase.from('content_items')
      .select('id, title, seo_score')
      .eq('user_id', userId)
      .eq('status', 'published')
      .gte('updated_at', startDate)
      .lt('updated_at', endDate),

    supabase.from('keywords')
      .select('id', { count: 'exact' })
      .eq('user_id', userId)
      .gte('created_at', startDate)
      .lt('created_at', endDate),

    supabase.from('ai_strategy_proposals')
      .select('id', { count: 'exact' })
      .eq('user_id', userId)
      .eq('status', 'scheduled')
      .gte('updated_at', startDate)
      .lt('updated_at', endDate),

    supabase.from('content_calendar')
      .select('id', { count: 'exact' })
      .eq('user_id', userId)
      .eq('status', 'completed')
      .gte('scheduled_date', startDate)
      .lt('scheduled_date', endDate)
  ]);

  const allCreated = created.data || [];
  const avgSeo = allCreated.length
    ? Math.round(allCreated.reduce((sum: number, c: any) => sum + (c.seo_score || 0), 0) / allCreated.length)
    : 0;

  const top3 = [...allCreated]
    .sort((a: any, b: any) => (b.seo_score || 0) - (a.seo_score || 0))
    .slice(0, 3);

  const createdCount = created.count || 0;
  const publishedCount = published.data?.length || 0;
  const keywordsCount = keywords.count || 0;
  const proposalsCount = proposals.count || 0;
  const calendarCount = calendar.count || 0;

  // Build a readable summary
  let summary = `**${targetMonth} Monthly Summary**\n\n`;
  summary += `| Metric | Count |\n|--------|-------|\n`;
  summary += `| Content created | ${createdCount} |\n`;
  summary += `| Content published | ${publishedCount} |\n`;
  summary += `| Avg SEO score | ${avgSeo}/100 |\n`;
  summary += `| Keywords added | ${keywordsCount} |\n`;
  summary += `| Proposals acted on | ${proposalsCount} |\n`;
  summary += `| Calendar items completed | ${calendarCount} |\n`;

  if (top3.length > 0) {
    summary += `\n**Top content this month:**\n`;
    top3.forEach((c: any, i: number) => {
      summary += `${i + 1}. "${c.title}" — SEO ${c.seo_score || 0}/100\n`;
    });
  }

  if (createdCount === 0) {
    summary += `\nNo content created this month yet. Would you like to generate some proposals?`;
  }

  return JSON.stringify({
    success: true,
    month: targetMonth,
    message: summary,
    data: {
      content_created: createdCount,
      content_published: publishedCount,
      avg_seo_score: avgSeo,
      keywords_added: keywordsCount,
      proposals_acted: proposalsCount,
      calendar_completed: calendarCount,
      top_content: top3.map((c: any) => ({ title: c.title, seo_score: c.seo_score }))
    }
  });
}
```
