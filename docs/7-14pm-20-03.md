// Add near the other content tool handlers

async function handleImproveContent(toolArgs: any, userId: string, supabase: any, callAiProxy: Function) {
  // 1. Fetch the content
  const { data: content, error } = await supabase
    .from('content_items')
    .select('*')
    .eq('id', toolArgs.content_id)
    .eq('user_id', userId)
    .single();

  if (error || !content) {
    return JSON.stringify({ success: false, message: 'Content not found or access denied.' });
  }

  // 2. Snapshot current version before changing anything
  try {
    await supabase.from('content_versions').insert({
      content_id: content.id,
      user_id: userId,
      title: content.title,
      content: content.content,
      meta_title: content.meta_title,
      meta_description: content.meta_description,
      seo_score: content.seo_score,
      change_source: 'ai_improvement',
      change_description: `Before improvement: ${toolArgs.instruction}`
    });
  } catch { /* version save is non-blocking */ }

  // 3. Build the improvement prompt
  const scope = toolArgs.scope || 'full';
  const prompt = `You are improving an existing article. Here is the current content:

TITLE: ${content.title}
KEYWORD: ${content.main_keyword || 'not set'}
CURRENT SEO SCORE: ${content.seo_score || 'not scored'}

CONTENT:
${content.content}

INSTRUCTION: ${toolArgs.instruction}
SCOPE: ${scope}

Rules:
- Maintain the same voice, style, and structure unless told otherwise
- Keep all existing internal links intact
- If improving SEO: ensure the keyword appears in the first 100 words, improve headings, add FAQ section if missing
- If shortening: preserve all key points and structure, remove filler and redundancy
- If expanding: add real value (examples, data, actionable tips), not fluff or padding
- Return ONLY the improved HTML content. No commentary, no explanations — just the HTML.`;

  // 4. Call AI
  const aiResponse = await callAiProxy(supabase, userId, {
    messages: [{ role: 'user', content: prompt }],
    max_tokens: 4000
  });

  const improvedContent = aiResponse.choices?.[0]?.message?.content;
  if (!improvedContent) {
    return JSON.stringify({ success: false, message: 'AI did not return improved content. Try again.' });
  }

  // 5. Rescore SEO and save
  const newScore = calculateBasicSeoScore(improvedContent, content.main_keyword, content.meta_title, content.meta_description);

  await supabase.from('content_items').update({
    content: improvedContent,
    seo_score: newScore,
    updated_at: new Date().toISOString()
  }).eq('id', content.id).eq('user_id', userId);

  const scoreDelta = newScore - (content.seo_score || 0);

  return JSON.stringify({
    success: true,
    message: `Improved "${content.title}". SEO: ${content.seo_score || 0} → ${newScore} (${scoreDelta >= 0 ? '+' : ''}${scoreDelta}). Previous version saved — restore anytime.`,
    content_id: content.id,
    old_score: content.seo_score,
    new_score: newScore
  });
}
```

// Generate meta title and description via AI instead of truncation
let autoMetaTitle = autoTitle;
let autoMetaDesc = '';

try {
  const metaResponse = await callAiProxyWithRetry(supabase, userId, {
    messages: [{
      role: 'user',
      content: `Generate a meta title and meta description for this article.

TITLE: ${autoTitle}
KEYWORD: ${toolArgs.keyword || ''}
FIRST 500 CHARS: ${plainText.substring(0, 500)}

Rules for meta TITLE (max 60 characters):
- Include the primary keyword near the start
- Use a power word (Ultimate, Complete, Essential, Proven, Simple)
- Be specific — include a number or year if natural

Rules for meta DESCRIPTION (max 155 characters):
- Write a HOOK that creates curiosity, not a summary
- Start with an action verb or bold claim
- Include the keyword naturally
- End with implied value ("here's how", "that actually work", "you need to know")

Return ONLY valid JSON: {"meta_title": "...", "meta_description": "..."}`
    }],
    max_tokens: 200
  });

  const parsed = JSON.parse(metaResponse.choices?.[0]?.message?.content || '{}');
  if (parsed.meta_title && parsed.meta_title.length <= 60) autoMetaTitle = parsed.meta_title;
  if (parsed.meta_description && parsed.meta_description.length <= 155) autoMetaDesc = parsed.meta_description;
} catch {
  // Fallback to truncation if AI call fails
}

// Final fallback
if (!autoMetaDesc) {
  autoMetaDesc = plainText.length > 155 ? plainText.substring(0, 152) + '...' : plainText;
}
if (!autoMetaTitle || autoMetaTitle.length > 60) {
  autoMetaTitle = autoTitle.length > 60 ? autoTitle.substring(0, 57) + '...' : autoTitle;
}

async function handleMonthlySummary(toolArgs: any, userId: string, supabase: any) {
  const targetMonth = toolArgs.month || new Date().toISOString().slice(0, 7);
  const startDate = `${targetMonth}-01`;
  const nextMonth = new Date(startDate);
  nextMonth.setMonth(nextMonth.getMonth() + 1);
  const endDate = nextMonth.toISOString().slice(0, 10);

  const [created, published, keywords, proposals, calendar] = await Promise.all([
    supabase.from('content_items').select('id, title, seo_score, status', { count: 'exact' })
      .eq('user_id', userId).gte('created_at', startDate).lt('created_at', endDate),
    supabase.from('content_items').select('id, title, seo_score')
      .eq('user_id', userId).eq('status', 'published').gte('updated_at', startDate).lt('updated_at', endDate),
    supabase.from('keywords').select('id', { count: 'exact' })
      .eq('user_id', userId).gte('created_at', startDate).lt('created_at', endDate),
    supabase.from('ai_strategy_proposals').select('id', { count: 'exact' })
      .eq('user_id', userId).eq('status', 'scheduled').gte('updated_at', startDate).lt('updated_at', endDate),
    supabase.from('content_calendar').select('id', { count: 'exact' })
      .eq('user_id', userId).eq('status', 'completed').gte('scheduled_date', startDate).lt('scheduled_date', endDate)
  ]);

  const avgSeo = created.data?.length
    ? Math.round(created.data.reduce((sum: number, c: any) => sum + (c.seo_score || 0), 0) / created.data.length)
    : 0;

  const top3 = [...(created.data || [])].sort((a: any, b: any) => (b.seo_score || 0) - (a.seo_score || 0)).slice(0, 3);

  return JSON.stringify({
    success: true,
    month: targetMonth,
    summary: {
      content_created: created.count || 0,
      content_published: published.data?.length || 0,
      avg_seo_score: avgSeo,
      keywords_added: keywords.count || 0,
      proposals_acted: proposals.count || 0,
      calendar_completed: calendar.count || 0,
      top_content: top3.map((c: any) => ({ title: c.title, seo_score: c.seo_score }))
    },
    message: `**${targetMonth} Summary:** ${created.count || 0} articles created, ${published.data?.length || 0} published, avg SEO ${avgSeo}/100, ${keywords.count || 0} keywords added, ${proposals.count || 0} proposals acted on.`
  });
}
```
