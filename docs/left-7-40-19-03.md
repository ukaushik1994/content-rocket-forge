```ts
let performanceInsight = '';
try {
  const { data: scored } = await supabase
    .from('content_items')
    .select('main_keyword, seo_score')
    .eq('user_id', user.id)
    .not('seo_score', 'is', null)
    .order('seo_score', { ascending: false })

    const themes: Record<string, { total: number; count: number }> = {};
    for (const c of scored) {
      const theme = (c.main_keyword || '').split(/\s+/)[0]?.toLowerCase();
      if (!theme || theme.length < 3) continue;
      if (!themes[theme]) themes[theme] = { total: 0, count: 0 };
      themes[theme].total += c.seo_score || 0;
      themes[theme].count++;
    }
    const ranked = Object.entries(themes)
      .filter(([_, v]) => v.count >= 2)
      .map(([k, v]) => ({ theme: k, avg: Math.round(v.total / v.count) }))
      .sort((a, b) => b.avg - a.avg)
      .slice(0, 5);

    if (ranked.length >= 2) {
      performanceInsight = `\n\n## CONTENT PERFORMANCE BY TOPIC\n${ranked.map(t => `- "${t.theme}" topics: avg SEO ${t.avg}/100`).join('\n')}\nRecommend more content in high-performing areas.`;

    }
```

**Frontend:** No changes needed — the message response will include the warning.

**Backend:** `supabase/functions/enhanced-ai-chat/content-action-tools.ts` — inside `create_calendar_item`, after the successful insert, before the return.

```ts
// After the insert succeeds (around the return statement):
let diversityNote = '';
try {
  const monthStart = new Date();
  monthStart.setDate(1);
  const monthEnd = new Date(monthStart);
  monthEnd.setMonth(monthEnd.getMonth() + 1);

  const { data: monthItems } = await supabase
    .from('content_calendar')
    .select('content_type')
    .eq('user_id', userId)
    .gte('scheduled_date', monthStart.toISOString().split('T')[0])
    .lt('scheduled_date', monthEnd.toISOString().split('T')[0]);

  if (monthItems && monthItems.length >= 3) {
    const types: Record<string, number> = {};
    for (const item of monthItems) {
      const t = item.content_type || 'blog';
      types[t] = (types[t] || 0) + 1;
    }
    const [dominant, count] = Object.entries(types).sort((a, b) => b[1] - a[1])[0];
    const pct = Math.round(count / monthItems.length * 100);
    if (pct >= 70) {
      diversityNote = ` (Note: ${pct}% of this month's calendar is ${dominant} — consider diversifying formats.)`;
    }
  }
} catch (e) { /* non-blocking */ }

return { success: true, message: `Scheduled "${data.title}" for ${data.scheduled_date}${diversityNote}`, item: data };
```
